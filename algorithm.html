Search Algorithm

BEGIN SearchOutfits(query, outfits) INPUT query FROM user INPUT outfits list results = [] FOR EACH outfit IN outfits IF outfit.title CONTAINS query OR ANY tag IN outfit.tags CONTAINS query ADD outfit TO results END IF END FOR RETURN results END SearchOutfits
def search_outfits(query, outfits): results = [] query = query.lower() for outfit in outfits: if query in outfit['title'].lower(): results.append(outfit) else: for tag in outfit['tags']: if query in tag.lower(): results.append(outfit) break return results

Sort Algorithm

BEGIN SortOutfitsByRating(outfits) n = LENGTH(outfits) FOR i = 0 TO n-1 FOR j = 0 TO n-i-2 IF outfits[j].rating < outfits[j+1].rating SWAP outfits[j] AND outfits[j+1] END IF END FOR END FOR RETURN outfits END SortOutfitsByRating
def sort_outfits_by_rating(outfits): n = len(outfits) for i in range(n): for j in range(0, n-i-1): if outfits[j]['rating'] < outfits[j+1]['rating']: outfits[j], outfits[j+1] = outfits[j+1], outfits[j] return outfits

Filter Algorithm

BEGIN FilterOutfits(outfits, style, minRating) filtered = [] FOR EACH outfit IN outfits IF (outfit.style == style OR style == "All") AND outfit.rating >= minRating ADD outfit TO filtered END IF END FOR RETURN filtered END FilterOutfits
def filter_outfits(outfits, style, min_rating): filtered = [] for outfit in outfits: if (outfit['style'] == style or style == "All") and outfit['rating'] >= min_rating: filtered.append(outfit) return filtered

Average Rating

BEGIN CalculateAverageRating(outfits) total = 0 count = 0 FOR EACH outfit IN outfits total = total + outfit.rating count = count + 1 END FOR IF count > 0 THEN average = total / count RETURN average ELSE RETURN 0 END IF END CalculateAverageRating
def calculate_average_rating(outfits): if not outfits: return 0 total = sum(outfit['rating'] for outfit in outfits) return total / len(outfits)

Recommendation Algorithm

BEGIN RecommendOutfits(outfits, userPreferences) recommendations = [] FOR EACH outfit IN outfits score = 0 IF outfit.season IN userPreferences.seasons score = score + 2 END IF IF outfit.rating >= 4.5 score = score + 1 END IF IF ANY tag IN outfit.tags IN userPreferences.tags score = score + 1 END IF IF score >= 2 ADD outfit TO recommendations END IF END FOR RETURN recommendations END RecommendOutfits
def recommend_outfits(outfits, user_preferences): recommendations = [] for outfit in outfits: score = 0 if outfit['style'] in user_preferences['style']: score += 2 if outfit['rating'] >= 4.5: score += 1 for tag in outfit['tags']: if tag in user_preferences['tags']: score += 1 break if score >= 2: recommendations.append(outfit) return recommendations

Trending Algorithm

BEGIN FindTrendingOutfits(outfits) trending = [] FOR EACH outfit IN outfits IF outfit.rating >= 4.7 AND outfit.views > 1000 ADD outfit TO trending END IF END FOR SORT trending BY rating DESC RETURN trending END FindTrendingOutfits
def find_trending_outfits(outfits): trending = [] for outfit in outfits: if outfit['rating'] >= 4.7 and outfit['views'] > 1000: trending.append(outfit) # Sort by rating descending trending.sort(key=lambda x: x['rating'], reverse=True) return trending

Shuffle Algorithm

BEGIN ShuffleOutfits(outfits) n = LENGTH(outfits) FOR i = n-1 DOWNTO 1 j = RANDOM INTEGER BETWEEN 0 AND i SWAP outfits[i] AND outfits[j] END FOR RETURN outfits END ShuffleOutfits
import random def shuffle_outfits(outfits): shuffled = outfits.copy() n = len(shuffled) for i in range(n-1, 0, -1): j = random.randint(0, i) shuffled[i], shuffled[j] = shuffled[j], shuffled[i] return shuffled

Pagination Algorithm

BEGIN PaginateOutfits(outfits, page, itemsPerPage) startIndex = (page - 1) * itemsPerPage endIndex = startIndex + itemsPerPage IF startIndex >= LENGTH(outfits) RETURN [] END IF IF endIndex > LENGTH(outfits) endIndex = LENGTH(outfits) END IF pageOutfits = outfits[startIndex:endIndex] RETURN pageOutfits END PaginateOutfits
def paginate_outfits(outfits, page, items_per_page): start_index = (page - 1) * items_per_page end_index = start_index + items_per_page if start_index >= len(outfits): return [] if end_index > len(outfits): end_index = len(outfits) return outfits[start_index:end_index]

Duplicate Check

BEGIN FindDuplicates(outfits) seen = {} duplicates = [] FOR EACH outfit IN outfits key = outfit.title + JOIN(outfit.tags) IF key IN seen ADD outfit TO duplicates ELSE seen[key] = True END IF END FOR RETURN duplicates END FindDuplicates
def find_duplicates(outfits): seen = set() duplicates = [] for outfit in outfits: key = outfit['title'] + ''.join(outfit['tags']) if key in seen: duplicates.append(outfit) else: seen.add(key) return duplicates
